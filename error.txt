
from TM1py import TM1Service
import configparser
import json, os

config = configparser.ConfigParser()
config.read(r".\config.ini")   # <-- also fixed the SyntaxWarning (raw string)
publication_path     = config.get('Publication', 'publication_folder')
publication_year     = config.get('Publication', 'publication_year')
month_number         = config.get('Publication', 'month_number')
current_month_folder = publication_year + month_number

server_name    = 'ABSA Bank'

EXCLUDE_PREFIXES = (
    '}Stats', '}Clients', '}Groups', '}ClientGroups',
    '}ClientProperties', '}Capabilities', '}CapabilityAssignments',
    '}ElementAttributes_}',
)

with TM1Service(**config['BBPS_Development']) as tm1_dev:
    print(f'{server_name}: logged in')

    backup_location ='Update Production/model'
    os.makedirs('Update Production/model/dimensions', exist_ok=True)
    os.makedirs('Update Production/model/cubes', exist_ok=True)
    os.makedirs('Update Production/model/rules', exist_ok=True)
    os.makedirs('Update Production/model/processes', exist_ok=True)

    # Dimensions
    for dim_name in tm1_dev.dimensions.get_all_names():
        dim = tm1_dev.dimensions.get(dim_name)
        with open(f'Update Production/model/dimensions/{dim_name}.json', 'w') as f:
            json.dump(json.loads(dim.body), f, indent=2)

    # Cubes
    for cube_name in tm1_dev.cubes.get_all_names():
        cube = tm1_dev.cubes.get(cube_name)
        with open(f'Update Production/model/cubes/{cube_name}.json', 'w') as f:
            json.dump(json.loads(cube.body), f, indent=2)

    # Processes
    for process_name in tm1_dev.processes.get_all_names():
        process = tm1_dev.processes.get(process_name)
        with open(f'Update Production/model/processes/{process_name}.json', 'w') as f:
            json.dump(json.loads(process.body), f, indent=2)
    
    # --- Chores 
    os.makedirs('Update Production/model/chores', exist_ok=True)
    for chore_name in tm1_dev.chores.get_all_names():
        chore = tm1_dev.chores.get(chore_name)
        with open(f'Update Production/model/chores/{chore_name}.json', 'w') as f:
            f.write(chore.body)
    
    # --- Public Subsets (skip private — those belong to users) ---
    for dim_name in tm1_dev.dimensions.get_all_names():
        if any(dim_name.startswith(p) for p in EXCLUDE_PREFIXES):
            continue
        os.makedirs(f'Update Production/model/subsets/{dim_name}', exist_ok=True)
        subset_names = tm1_dev.subsets.get_all_names(
            dimension_name=dim_name,
            hierarchy_name=dim_name,
            private=False                    # ◄ public only
        )
        for subset_name in subset_names:
            subset = tm1_dev.subsets.get(
                subset_name=subset_name,
                dimension_name=dim_name,
                hierarchy_name=dim_name,
                private=False
            )
            with open(f'Update Production/model/subsets/{dim_name}/{subset_name}.json', 'w') as f:
                f.write(subset.body)

    # --- Public Views (skip private — those belong to users) ---
    for cube_name in tm1_dev.cubes.get_all_names():
        if any(cube_name.startswith(p) for p in EXCLUDE_PREFIXES):
            continue
        os.makedirs(f'Update Production/model/views/{cube_name}', exist_ok=True)
        view_names = tm1_dev.views.get_all_names(
            cube_name=cube_name,
            private=False                    # ◄ public only
        )
        for view_name in view_names:
            view = tm1_dev.views.get(
                view_name=view_name,
                cube_name=cube_name,
                private=False
            )
            with open(f'Update Production/model/views/{cube_name}/{view_name}.json', 'w') as f:
                f.write(view.body)
